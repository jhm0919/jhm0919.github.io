Online compiler 에서 C++언어로 Hello world 출력해보기

#include <iostream>
using namespace std;
int main()
{
	cout << "Hello World" << endl;
	return 0;
}



#include <iostream>
int main()
{
	std::cout << "Hello World!\n";
}
